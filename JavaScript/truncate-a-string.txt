function truncateString (str, num) {
  if (str.length > num) {
    // let strHead = str.slice(0, num);
    // let replace = ".".repeat(3);
    // return strHead + replace
    return str.replace(str.slice(num), "...")
  }
  else {
    return str
  }
}
